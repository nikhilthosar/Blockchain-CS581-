\element{class3}{
  \begin{question}{c3q1}
   What happens if a single leaf node's data in a Merkle Tree is altered?
    \begin{choices}
      \wrongchoice{Only that leaf node's hash changes}
      \wrongchoice{No change occurs in the tree structure}
      \wrongchoice{All hashes above it remain unaffected}
      \correctchoice{The root hash and subsequent hashes above it will change}
    \end{choices}
  \end{question}
}

\element{class3}{
  \begin{question}{c3q2}
    How does a Merkle Tree contribute to data integrity in a blockchain?
    \begin{choices}
      \wrongchoice{By encrypting all data in the blockchain}
      \wrongchoice{By storing all transactions in a single block}
      \wrongchoice{By making it easier to verify the authenticity of individual transactions}
      \correctchoice{By providing a tamper-evident structure for the entire dataset}
    \end{choices}
  \end{question}
}
